Up and Running in 60 Seconds

# Install: pip install agenticai
import agenticai

# 1. Authenticate with your API key
agenticai.configure(api_key="sk-live-xxxxxxxxxxxxxxxx")

# 2. Describe your pipeline in plain English
result = agenticai.run(
    task="Load CSV from S3, filter revenue > 10000, group by region, chart",
    data_source="s3://my-bucket/sales_data.csv",
    custom_functions="./my_transforms.py",  # optional
)

# 3. Use the result
print(result.summary)
result.chart.save("revenue_by_region.png")
